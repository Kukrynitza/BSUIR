#!/bin/bash

if [[ "$#" -ne 2 ]]; then
    echo "Неправильно введены данные, пример: (анализируемый_каталог файл_для_записи)"
    exit 1
fi

direct=$1
out=$2

if [[ ! -d "$direct" ]]; then
    echo "Ошибка: '$direct' не является каталогом."
fi

if [[ ! -w "$out" ]]; then
    echo "Ошибка: '$direct' не является каталогом."
fi

size=0
count=0

count_size() {
    local dir=$1
    for file in "$dir"/*; do
        if [[ -d "$file" ]]; then
            count=$((count + 1))
            count_size "$file"
        elif [[ -f "$file" ]]; then
            file_size=$(stat -c%s "$file")
            size=$((size + file_size))
            count=$((count + 1))
        fi
    done
}

count_size "$direct"

size_mb=$(echo "scale=2; $size / 1024 / 1024" | bc)
size_kb=$(echo "scale=2; $size / 1024" | bc)

echo "$(realpath "$direct"), ${size_mb} MB, ${size_kb} KB, $count штук" > "$out"

echo "Результаты сохранены в '$out'."
